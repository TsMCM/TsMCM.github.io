鸡兔同笼

鸡兔同笼，共35只头，94只脚，问鸡兔各多少？

Python 演示代码

heads = 35  # 头的数量
feet = 94   # 脚的数量

for chickens in range(feet//2 + 1):
   total_feet = chickens * 2 + (heads - chickens) * 4
   if total_feet == feet:
       print(f"鸡的数量：{chickens}")
       print(f"兔子的数量：{(heads - chickens)}")
       break

这段代码首先定义了头的数量（heads）和脚的数量（feet）。然后，我们使用一个for循环来枚举鸡的数量，其中鸡的数量不能超过头的数量。在循环内部，我们计算鸡和兔子的总脚数（total_feet），如果总脚数等于脚的数量，就输出鸡和兔子的数量，并跳出循环。

Go 演示代码

package main

import (
    "bufio"
    "fmt"
    "os"
)

func main() 
{
    reader := bufio.NewReader(os.Stdin)
    fmt.Print("请输入头数：")
    text, _ := reader.ReadString('\n')
    heads, _ := strconv.Atoi(text[:len(text)-1])
    fmt.Print("请输入脚数：")
    text, _ = reader.ReadString('\n')
    feet, _ := strconv.Atoi(text[:len(text)-1])
    chickens, rabbits := solveChickenRabbitProblem(heads, feet)
    fmt.Printf("鸡的数量：%d\n兔的数量：%d\n", chickens, rabbits)
}

func solveChickenRabbitProblem(heads, feet int) (chickens, rabbits int) 
{
    for chickens < heads && (chickens+rabbits)*2 <= feet 
    {
        chickens++
        rabbits = heads - chickens
    }
    return
}

百钱百鸡

100元购买100只鸡，公鸡5元一只，母鸡3元一只，小鸡1元三只，如果花完100元，最后能买回多少只公鸡、母鸡、小鸡？

Python 演示代码

price_chickens = 100  # 100元
total_chickens = 100  # 购买100只鸡

# 假设公鸡的数量为x，母鸡的数量为y，小鸡的数量为z
# 则有：
# x * 5 + y * 3 + z * 1.0/3 = price_chickens
# x + y + z = total_chickens

# 解这个方程组得到：
x = int((price_chickens - total_chickens) / 5)
y = int((price_chickens - total_chickens) / 3)
z = int(total_chickens - x - y)

print("公鸡：", x, "只")
print("母鸡：", y, "只")
print("小鸡：", z, "只")

Zig 演示代码

const std = @import("std");

pub fn main() void 
{
   var price_chickens = 100; // 100元
   var total_chickens = 100; // 购买100只鸡

   var x = solve(price_chickens, total_chickens, 5);
   var y = solve(price_chickens, total_chickens, 3);
   var z = total_chickens - x - y;

   std.debug.print("公鸡：{}只\n", .{x});
   std.debug.print("母鸡：{}只\n", .{y});
   std.debug.print("小鸡：{}只\n", .{z});
}

pub fn solve(price_chickens: i32, total_chickens: i32, cost: i32) i32 
{
   if (price_chickens == 0) 
   {
       return 0;
   }
   var chickens = total_chickens - 1;
   var cost_per_chicken = cost;
   var remaining_price = price_chickens - (chickens * cost_per_chicken);
   if (remaining_price >= 0) 
   {
       return chickens + solve(remaining_price, total_chickens, cost);
   } 
   else 
   {
       return 0;
   }
}

这个程序首先定义了购买鸡的总金额和鸡的总数量，然后通过递归函数来计算公鸡、母鸡和小鸡的数量。最后，使用std.debug.print函数输出结果。

约瑟夫环

在某一款游戏中，曾有一个幸运儿的头衔，需要进入指定房间，完成任务即可获得。 任务规则如下：

例如：现在进入了 六名玩家，所有人需要坐在围成一个圆的凳子上，被碰到的玩家视为淘汰。

给他们标记为 1-6 号 1碰2、3碰4、5碰6; 1碰3、5碰1、5获得。

现在假设共有38名玩家进入该房间开始游戏，这些人之中，有一个叫 收藏者 的玩家，我们应该让他坐在哪个位置，才能获得该头衔。

Python 演示代码

def josephus(n, m):
    people = list(range(1, n + 1))
    index = 0
    while len(people) > 1:
        index = (index + m - 1) % len(people)
        del people[index]
    return people[0]

n = 38
m = 2
result = josephus(n, m)
print("收藏者的位置：", result)

C语言 演示代码

#include <stdio.h>
#include <stdlib.h>

int josephus(int n, int m) 
{
   int *people = (int *) malloc(n * sizeof(int));
   if (people == NULL) 
   {
       printf("Error: malloc failed\n");
       exit(1);
   }
   for (int i = 0; i < n; i++) 
   {
       people[i] = i + 1;
   }
   int index = 0;
   while (n > 1) 
   {
       index = (index + m - 1) % n;
       free(people[index]);
       n--;
   }
   return people[0];
}

int main() 
{
   int n = 38;
   int m = 2;
   int result = josephus(n, m);
   printf("收藏者的位置：%d\n", result);
   return 0;
}

解释圆周率

Python 演示代码

import math

# 计算圆周率
pi = math.pi

# 输出圆周率
print("圆周率（π）的值为：", pi)

这段代码首先导入了 Python 的 math 模块，然后使用该模块中的 pi 函数计算圆周率。最后，我们通过 print 函数输出计算得到的圆周率。

C++ 演示代码

#include <iostream>
using namespace std;

int main() 
{
   int sum = 0; // 用于存储循环计算结果
   int i; // 循环计数器

   for (i = 1; i <= 1000000; i++) // 循环1000000次
   { 
       if (i % 2 == 1) 
       { // 如果i是奇数
           sum += 4.0 * (1.0 / i); // 累加4/i
       } 
       else // 如果i是偶数
       { 
           sum -= 4.0 * (1.0 / i); // 减去4/i
       }
   }

   cout << "圆周率约为：" << sum << endl; // 输出结果

   return 0;
}
